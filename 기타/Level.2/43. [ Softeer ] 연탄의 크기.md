### [ 문제 풀이 ]
```JavaScript
const fs = require("fs");
const input = fs.readFileSync(0).toString().trim().split("\n").map((line) => line.split(" ").map(Number));

const n = input[0];
const stove = input[1];

let answer = 0;

for(let i = 2; i <= 100; i++) {
    let cnt = 0;
    for(let j = 0; j < n; j++) {
        if(stove[j] % i === 0) cnt += 1;
    }
    answer = Math.max(answer, cnt);
}

console.log(answer);
```
